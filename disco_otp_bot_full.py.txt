# ====================== DATABASE ======================
def init_db():
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS users (
            user_id INTEGER PRIMARY KEY,
            phone TEXT,
            subscribed BOOLEAN DEFAULT 0,
            sub_expires REAL DEFAULT 0,
            card_issuer TEXT DEFAULT NULL,
            sms_enabled BOOLEAN DEFAULT 0
        )
    ''')
    conn.commit()
    conn.close()

def get_user(user_id):
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("SELECT phone, subscribed, sub_expires, card_issuer, sms_enabled FROM users WHERE user_id = ?", (user_id,))
    row = cursor.fetchone()
    conn.close()
    if row:
        return {"phone": row[0], "subscribed": bool(row[1]), "sub_expires": row[2], "card_issuer": row[3], "sms_enabled": bool(row[4])}
    return {"phone": None, "subscribed": False, "sub_expires": 0, "card_issuer": None, "sms_enabled": False}

def save_user(user_id, phone=None, subscribed=None, sub_expires=None, card_issuer=None, sms_enabled=None):
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("SELECT 1 FROM users WHERE user_id = ?", (user_id,))
    exists = cursor.fetchone()
    if exists:
        updates, params = [], []
        if phone is not None: updates.append("phone = ?"); params.append(phone)
        if subscribed is not None: updates.append("subscribed = ?"); params.append(1 if subscribed else 0)
        if sub_expires is not None: updates.append("sub_expires = ?"); params.append(sub_expires)
        if card_issuer is not None: updates.append("card_issuer = ?"); params.append(card_issuer)
        if sms_enabled is not None: updates.append("sms_enabled = ?"); params.append(1 if sms_enabled else 0)
        if updates:
            query = f"UPDATE users SET {', '.join(updates)} WHERE user_id = ?"
            params.append(user_id)
            cursor.execute(query, params)
    else:
        cursor.execute('''
            INSERT INTO users (user_id, phone, subscribed, sub_expires, card_issuer, sms_enabled)
            VALUES (?, ?, ?, ?, ?, ?)
        ''', (user_id, phone, 1 if subscribed else 0, sub_expires or 0, card_issuer, 1 if sms_enabled else 0))
    conn.commit()
    conn.close()

# ====================== SMS GATEWAY (Twilio) ======================
def send_sms_via_twilio(phone, message):
    if not all([TWILIO_ACCOUNT_SID, TWILIO_AUTH_TOKEN, TWILIO_PHONE_NUMBER]):
        return False, "SMS gateway not configured."
    try:
        from twilio.rest import Client
        client = Client(TWILIO_ACCOUNT_SID, TWILIO_AUTH_TOKEN)
        msg = client.messages.create(body=message, from_=TWILIO_PHONE_NUMBER, to=phone)
        return True, f"SMS sent (SID: {msg.sid})"
    except Exception as e:
        return False, f"SMS failed: {str(e)}"

@bot.message_handler(commands=['code'])
def send_otp(message):
    user_id = message.from_user.id
    user = get_user(user_id)
    if not user["subscribed"] or user["sub_expires"] < time.time():
        bot.reply_to(message, "❌ Premium required or expired.")
        return
    if not user["phone"]:
        bot.reply_to(message, "❌ No phone on file. Use /phone first.")
        return
    if not user["card_issuer"]:
        bot.reply_to(message, "❌ No card selected. Use /discover, /bestbuy, etc.")
        return

    code = random.randint(100000, 999999)
    template = CARD_TEMPLATES.get(user["card_issuer"], CARD_TEMPLATES["discover"])
    otp_message = template.format(phone=user["phone"], code=code)

    if user.get("sms_enabled"):
        success, result = send_sms_via_twilio(user["phone"], otp_message)
        if success:
            bot.reply_to(message, f"✅ OTP sent via real SMS!\n{result}")
        else:
            bot.reply_to(message, f"⚠️ SMS failed.\n\n{otp_message}")
    else:
        bot.reply_to(message, otp_message)
